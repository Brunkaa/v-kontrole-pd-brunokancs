public class Main {
    public static void main(String[] args) {
        int n = 100;
        long[] fib = new long[n];

        // pirmie divi Fibonači skaitļi
        fib[0] = 0;
        fib[1] = 1;

        // ģenerē pirmos 100 Fibonači skaitļus
        for (int i = 2; i < n; i++) {
            fib[i] = fib[i - 1] + fib[i - 2];
        }

        // izvada apgrieztā secībā
        for (int i = n - 1; i >= 0; i--) {
            System.out.print(fib[i] + " ");
        }
    }
}
